#include <stdio.h>

int main(void) {
  int n1 = 0, n2 = 0, n3 = 0;
  printf ("Digite o primeiro número: ");
  scanf ("%d", &n1);
  printf("Digite o segundo número: ");
  scanf ("%d", &n2);
  printf("Digite o terceiro número: ");
  scanf ("%d", &n3);
  if (n1>(n2+n3)){
    printf ("----------------------- \n");
    printf ("Primeiro número: %d \n", n1);
    printf ("Soma dos outros dois numeros: %d \n", n2+n3);
    printf("Então, o primeiro número indicado é maior que a soma dos outros dois \n");

  } else if (n1 == (n2+n3)) {
    printf ("----------------------- \n");
    printf ("Primeiro número: %d \n", n1);
    printf ("Soma dos outros dois numeros: %d \n", n2+n3);
    printf("Então, o primeiro número indicado é igual que a soma dos outros dois \n");
  } else {
    printf ("----------------------- \n");
    printf ("Primeiro número: %d \n", n1);
    printf ("Soma dos outros dois numeros: %d \n", n2+n3);
    printf("Então, o primeiro número indicado é menor que a soma dos outros dois \n");
  }
  return 0;
}
